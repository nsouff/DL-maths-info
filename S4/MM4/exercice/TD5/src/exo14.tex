\section{Exercice 14}
\subsection*{Question 1}
Montrons que $\lim\limits_{k \to +\infty} f_k(x_k) = f(x)$. Fixons $\varepsilon > 0$.
\begin{align*}
  \forall k \in \mathbb{N}~|f_k(x_k)-f(x)| &= |f_k(x_k)-f(x_k)+f(x_k)-f(x)|\\
                                           &\leq |f_k(x_k)-f(x_k)|+|f(x_k)-f(x)|
\end{align*}
